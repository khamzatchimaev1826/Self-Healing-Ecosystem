from sklearn.ensemble import RandomForestClassifier

class HealthClassifier:
    def __init__(self):
        self.model = RandomForestClassifier(
            n_estimators=100,
            max_depth=10,
            random_state=42
        )
    
    def train(self, X_train, y_train):
        self.model.fit(X_train, y_train)
    
    def predict(self, features):
        prediction = self.model.predict([features])[0]
        confidence = max(self.model.predict_proba([features])[0])
        return {"status": prediction, "confidence": round(confidence, 2)}

Model 4: Dynamic Simulation Engine (Lotka-Volterra Equations)
Purpose: Simulate real-time population dynamics using differential equationsMathematical Model:
plaintext
dx/dt = αx - βxy  (Prey population change)
dy/dt = δxy - γy  (Predator population change)

Where:
- x = prey population
- y = predator population
- α = prey growth rate
- β = predation rate
- δ = predator growth from prey
- γ = predator death rate
Implementation:
python
import numpy as np

class SimulationEngine:
    def __init__(self, ecosystem_data):
        self.plants = ecosystem_data['plant_population']
        self.herbivores = ecosystem_data['herbivores']
        self.predators = ecosystem_data['predators']
        self.temperature = ecosystem_data['temperature']
        self.rainfall = ecosystem_data['rainfall']
    
    def simulate_step(self, dt=0.1):
        # Plant growth (logistic)
        plant_growth = 0.5 * self.plants * (1 - self.plants / 100)
        plant_consumption = 0.02 * self.herbivores * self.plants
        self.plants += (plant_growth - plant_consumption) * dt
        
        # Herbivore dynamics
        herb_growth = 0.03 * self.plants * self.herbivores
        herb_death = 0.1 * self.herbivores * self.predators
        self.herbivores += (herb_growth - herb_death) * dt
        
        # Predator dynamics
        pred_growth = 0.02 * self.herbivores * self.predators
        pred_death = 0.15 * self.predators
        self.predators += (pred_growth - pred_death) * dt
        
        # Ensure non-negative populations
        self.plants = max(0, self.plants)
        self.herbivores = max(0, self.herbivores)
        self.predators = max(0, self.predators)
        
        return {
            'plants': round(self.plants, 2),
            'herbivores': round(self.herbivores, 2),
            'predators': round(self.predators, 2)
        }
    
    def calculate_health_score(self):
        # Weighted combination of factors
        biodiversity = min(1, (self.plants + self.herbivores + self.predators) / 150)
        temp_factor = 1 - abs(self.temperature - 25) / 25
        rain_factor = min(1, self.rainfall / 1000)
        
        health = (biodiversity * 0.4 + temp_factor * 0.3 + rain_factor * 0.3) * 100
        return round(min(100, max(0, health)), 2)
    
    def calculate_stability_index(self):
        total_pop = self.plants + self.herbivores + self.predators
        if total_pop == 0:
            return 0
        
        evenness = 1 - (abs(self.plants - self.herbivores) + 
                       abs(self.herbivores - self.predators)) / total_pop
        return round(max(0, min(1, evenness)), 2)
    
    def calculate_collapse_probability(self):
        health = self.calculate_health_score()
        if health > 70:
            return 0.05
        elif health > 40:
            return 0.3
        else:
            return 0.85

class RecoveryStrategyRecommender:
    def recommend(self, ecosystem_state):
        recommendations = []

        deforestation = ecosystem_state['deforestation_rate']
        biodiversity = ecosystem_state.get('biodiversity_index', 0.5)

import numpy as np
import pandas as pd

def generate_ecosystem_dataset(n_samples=10000):
    np.random.seed(42)
    
    # Generate features
    temperature = np.random.uniform(15, 45, n_samples)
    rainfall = np.random.uniform(200, 3000, n_samples)
    plant_population = np.random.uniform(10, 100, n_samples)
    herbivores = np.random.uniform(5, 100, n_samples)
    predators = np.random.uniform(2, 50, n_samples)
    pollution = np.random.uniform(0, 100, n_samples)
    deforestation_rate = np.random.uniform(0, 10, n_samples)
    
    # Calculate health scores based on realistic rules
    health_scores = []
    for i in range(n_samples):
        score = 100
        
        # Temperature impact (optimal: 20-30°C)
        if temperature[i] > 35:
            score -= (temperature[i] - 35) * 2
        elif temperature[i] < 20:
            score -= (20 - temperature[i]) * 1.5
        
        # Rainfall impact (optimal: 500-1500mm)
        if rainfall[i] < 500:
            score -= (500 - rainfall[i]) / 100 * 3
        elif rainfall[i] > 2000:
            score -= (rainfall[i] - 2000) / 100 * 2
        
        # Pollution impact
        if pollution[i] > 70:
            score -= (pollution[i] - 70) * 3
        
        # Deforestation impact
        score -= deforestation_rate[i] * 5
        
        # Biodiversity bonus
        if all([plant_population[i] > 30, herbivores[i] > 20, predators[i] > 5]):
            score += 10
        
        health_scores.append(max(0, min(100, score)))
    
    # Determine health status
    health_status = []
    for score in health_scores:
        if score >= 70:
            health_status.append('Healthy')
        elif score >= 40:
            health_status.append('Stressed')
        else:
            health_status.append('Collapsing')
    
    # Create DataFrame
    df = pd.DataFrame({
        'temperature': temperature,
        'rainfall': rainfall,
        'plant_population': plant_population,
        'herbivores': herbivores,
        'predators': predators,
        'pollution': pollution,
        'deforestation_rate': deforestation_rate,
        'health_score': health_scores,
        'health_status': health_status
    })
    
    return df

# Generate and save dataset
df = generate_ecosystem_dataset(10000)
df.to_csv('ecosystem_training_data.csv', index=False)
